create database GruappDta;

use GruappDta;

create table Clients (
id int auto_increment primary key,
name varchar (255),
email varchar (255),
password varchar (20)

);

create table Users (
id int auto_increment primary key,
name varchar (255),
email varchar (255),
phone varchar (20),
plate varchar (10),
type_car varchar (255),
ubication varchar (255)
);

SELECT * FROM Clients;
